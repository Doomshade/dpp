\documentclass[12pt, a4paper]{article}
\usepackage{amsmath}
\usepackage{syntax}
\newcommand{\ttt}[1]{
    \text{\texttt{#1}}
}
\newcommand{\ebnfnode}[2]{%
    #1\hspace{20px} &\to #2\ttt{ .}\\
}

\newcommand{\ebnfnodett}[2]{%
    \ebnfnode{\ttt{#1}}{\ttt{#2}}
}

\begin{document}
    \setlength{\grammarparsep}{0.6em} % increase separation between rules
    \setlength{\grammarindent}{10em} % increase separation between LHS/RHS

    \begin{grammar}

        <transl-unit> ::= \{ (<function> | <var-decl>) \}

        <function> ::= `func' <ident> <params> `->' <data-type> <block>

        <params> ::= `(' [<param> \{ `,' <param> \}] `)'

        <param> ::= <ident> `:' <data-type>

        <block> ::= `{' \{ <stat> \} `}'

        <stat> ::= `if' `(' <expr> `)' <stat> [`else' <stat>]
        \alt `for' `(' <ident> `to' <expr> `)' <stat>
        \alt `bye' <expr> `;'
        \alt `;'
        \alt <var-decl>
        \alt <block>
        \alt <expr> `;'

        <var-decl> ::= <data-type> `->' <ident> [`=' <expr>] `;'

        <data-type> ::= `xxlpp' // long
        \alt `pp' // int
        \alt `spp' // short
        \alt `xspp' // byte
        \alt `p' // char
        \alt `nopp' // void
        \alt `booba' // bool
        \alt `yarn' // string

        <expr> ::= <equ>

        <equ> ::= <comp> \{ (`!=' | `==') <comp> \}

        <comp> ::= <term> \{ (`>' | `>=' | `<=' | `<') <term> \}

        <term> ::= <factor> \{ (`+' | `-') <factor> \}

        <factor> ::= <unary> \{ (`*' | `/') <unary> \}

        <unary> ::= (`!' | `-') <unary> | <primary>

        <primary> ::= `yem' // true
        \alt `nom' // false
        \alt <number>
        \alt <string>
        \alt `(' <expr> `)'
        \alt <ident> [`(' [<args>] `)']

        <args> ::= <expr> \{ `,' <expr> \}
    \end{grammar}
    \begin{align*}
        \ebnfnodett{transl-unit}{( function | var-decl )*}
%        \ebnfnodett{struct-decl}{"struct" IDENTIFIER "\{" struct-field* "\}"}
%        \ebnfnodett{struct-field}{var-decl ";"}
        \ebnfnodett{function}{"FUNc" IDENT params "->" data-type block}
        \ebnfnodett{params}{"(" (param ("," param)*)? ")"}
        \ebnfnodett{param}{IDENT ":" data-type}
        \ebnfnodett{block}{"\{" stat* "\}"}
        \ebnfnode{\ttt{stat}}
        {%
            \ttt{"if" "(" expr ")" stat ( "else" stat )?}\\
            &\ttt{ | "for" "(" IDENT "to" expr ")" stat }\\
            &\ttt{ | "bye" expr ";" }\\
            &\ttt{ | ";" }\\
            &\ttt{ | var-decl }\\
            &\ttt{ | block }\\
            &\ttt{ | expr ";" }
        }
        \ebnfnodett{var-decl}{data-type "->" IDENT ( "=" expr )? ";"}
        \ebnfnode{\ttt{data-type}}
        {%
            \ttt{"xxlpp"}\\
            &\ttt{ | "pp"}\\
            &\ttt{ | "spp"}\\
            &\ttt{ | "xspp"}\\
            &\ttt{ | "p"}\\
            &\ttt{ | "nopp"}\\
            &\ttt{ | "booba"}\\
            &\ttt{ | "yarn"}
        }
        \ebnfnodett{expr}{equ}
        \ebnfnodett{equ}{comp ( ( "!=" | "==" ) comp )*}
        \ebnfnodett{comp}{term ( ( ">" | ">=" | "<" | "<=" ) term )*}
        \ebnfnodett{term}{factor ( ( "-" | "+" ) factor )*}
        \ebnfnodett{factor}{unary ( ( "/" | "*" ) unary )*}
        \ebnfnodett{unary}{( "!" | "-" ) unary | primary}
        \ebnfnode{\ttt{primary}}
        {%
            \ttt{"yem" | "nom" | NUMBER | STRING | "(" expr ")"}\\
            &\ttt{ | IDENT ( "(" ( args )? ")" )?}
        }
        \ebnfnodett{args}{"(" (expr ("," expr)*)? ")"}
    \end{align*}
\end{document}
