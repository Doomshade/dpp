\documentclass[12pt, a4paper]{article}
\usepackage{amsmath}
\usepackage{syntax}
\newcommand{\ttt}[1]{
    \text{\texttt{#1}}
}
\newcommand{\ebnfnode}[2]{%
    #1\hspace{20px} &\to #2\ttt{ .}\\
}

\newcommand{\ebnfnodett}[2]{%
    \ebnfnode{\ttt{#1}}{\ttt{#2}}
}

\begin{document}
    \setlength{\grammarparsep}{0.6em} % increase separation between rules
    \setlength{\grammarindent}{10em} % increase separation between LHS/RHS

    \begin{grammar}

        <transl-unit> ::= \{ (<function> | <var-decl>) \}

        <function> ::= `FUNc' <ident> <params> `->' <data-type> <block>

        <params> ::= `(' [<param> \{ `,' <param> \}] `)'

        <param> ::= <ident> `:' <data-type>

        <block> ::= `{' \{ <stat> \} `}'

        <stat> ::= `if' `(' <expr> `)' <stat> [`else' <stat>]
        \alt `for' `(' <ident> [ <expr> ] `to' <expr> `)' <stat>
        \alt `while' `(' <expr> `)' <stat>
        \alt `do' <block> `while' `(' <expr> `)' `;'
        \alt `loop' <block>
        \alt `break' `;'
        \alt `continue' `;'
        \alt `switch' `(' <expr> `)' `{' [<case> \{ `,' <case> \}] `}'
        \alt `bye' <expr> `;'
        \alt `pprint' `(' <expr> `)' `;'
        \alt `pprintln' `(' <expr> `)' `;'
        \alt `;'
        \alt <var-decl>
        \alt <block>
        \alt <expr> `;'

        <case> ::= `case' <expr> `:' <block>

        <var-decl> ::= <data-type> `->' <ident> `;'

        <data-type> ::= `pp' // int
        \alt `spp' // short
        \alt `xspp' // byte
        \alt `p' // char
        \alt `nopp' // void
        \alt `booba' // bool
        \alt `yarn' // string

        <expr> ::= <equ>

        <equ> ::= <comp> \{ (`!=' | `==') <comp> \}

        <comp> ::= <term> \{ (`>' | `>=' | `<=' | `<') <term> \}

        <term> ::= <factor> \{ (`+' | `-') <factor> \}

        <factor> ::= <unary> \{ (`*' | `/') <unary> \}

        <unary> ::= (`!' | `-') <unary> | <primary>

        <primary> ::= `yem' // true
        \alt `nom' // false
        \alt <number>
        \alt <string>
        \alt `(' <expr> `)'
        \alt <ident> [`(' [<args>] `)']

        <args> ::= <expr> \{ `,' <expr> \}
    \end{grammar}
\end{document}
